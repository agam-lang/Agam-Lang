Agam Language Formal Grammar

This document specifies the formal EBNF grammar for the Agam programming language. This grammar handles both base (indentation-sensitive) and advance (brace-delimited) syntax modes.

EBNF Reference

"rule-name" id="start">start"operator">: pragma* module

"rule-name" id="pragma">pragma"operator">: "@" IDENT ("." IDENT)* _NEWLINE

"rule-name" id="module">module"operator">: (_NEWLINE | declaration)*

"rule-name" id="declaration">declaration"operator">: visibility? annotation* decl_kind
"rule-name" id="decl_kind">decl_kind"operator">: function_decl
         | struct_decl
         | enum_decl
         | trait_decl
         | impl_decl
         | module_decl
         | use_decl
         | effect_decl
         | handler_decl
         | statement

"rule-name" id="visibility">visibility"operator">: "pub"

"rule-name" id="annotation">annotation"operator">: "@" IDENT ("." IDENT)*

"rule-name" id="function_decl">function_decl"operator">: "async"? "fn" IDENT LPAREN param_list? RPAREN ("->" type_expr)? body?
"rule-name" id="body">body"operator">: block | ":" block | ":" statement _NEWLINE

"rule-name" id="param_list">param_list"operator">: param ("," param)* ","?
"rule-name" id="param">param"operator">: pattern (":" type_expr)? ("=" expr)?

"rule-name" id="struct_decl">struct_decl"operator">: ("struct" | "class") IDENT struct_body
"rule-name" id="struct_body">struct_body"operator">: ":" _NEWLINE _INDENT field_decl* _DEDENT | LBRACE field_decl* RBRACE
"rule-name" id="field_decl">field_decl"operator">: visibility? IDENT ":" type_expr ("=" expr)? ","? _NEWLINE?

"rule-name" id="enum_decl">enum_decl"operator">: "enum" IDENT enum_body
"rule-name" id="enum_body">enum_body"operator">: ":" _NEWLINE _INDENT variant_decl* _DEDENT | LBRACE variant_decl* RBRACE
"rule-name" id="variant_decl">variant_decl"operator">: IDENT (LPAREN type_list RPAREN)? ","? _NEWLINE?

"rule-name" id="trait_decl">trait_decl"operator">: "trait" IDENT trait_body
"rule-name" id="trait_body">trait_body"operator">: ":" _NEWLINE _INDENT function_decl* _DEDENT | LBRACE function_decl* RBRACE

"rule-name" id="impl_decl">impl_decl"operator">: "impl" type_expr impl_body
"rule-name" id="impl_body">impl_body"operator">: ":" _NEWLINE _INDENT declaration* _DEDENT | LBRACE declaration* RBRACE

"rule-name" id="module_decl">module_decl"operator">: "mod" IDENT

"rule-name" id="use_decl">use_decl"operator">: ("use" | "import") path ("as" IDENT)?

"rule-name" id="effect_decl">effect_decl"operator">: "effect" IDENT effect_body
"rule-name" id="effect_body">effect_body"operator">: ":" _NEWLINE _INDENT effect_op* _DEDENT | LBRACE effect_op* RBRACE
"rule-name" id="effect_op">effect_op"operator">: "fn" IDENT LPAREN param_list? RPAREN ("->" type_expr)? ";" _NEWLINE?

"rule-name" id="handler_decl">handler_decl"operator">: "handle" IDENT "for" IDENT handler_body
"rule-name" id="handler_body">handler_body"operator">: ":" _NEWLINE _INDENT handler_clause* _DEDENT | LBRACE handler_clause* RBRACE
"rule-name" id="handler_clause">handler_clause"operator">: "fn" IDENT LPAREN ident_list? RPAREN ":" expr ";" _NEWLINE?

"rule-name" id="ident_list">ident_list"operator">: IDENT ("," IDENT)* ","?
"rule-name" id="type_list">type_list"operator">: type_expr ("," type_expr)* ","?

"rule-name" id="statement">statement"operator">: let_stmt
         | var_stmt
         | const_stmt
         | return_stmt
         | break_stmt
         | continue_stmt
         | while_stmt
         | for_stmt
         | loop_stmt
         | if_stmt
         | expr_stmt

"rule-name" id="stmt_end">stmt_end"operator">: (";" | _NEWLINE)+

"rule-name" id="let_stmt">let_stmt"operator">: "let" "mut"? pattern (":" type_expr)? ("=" expr)? stmt_end
"rule-name" id="var_stmt">var_stmt"operator">: "var" pattern (":" type_expr)? ("=" expr)? stmt_end
"rule-name" id="const_stmt">const_stmt"operator">: "const" IDENT (":" type_expr)? "=" expr stmt_end
"rule-name" id="return_stmt">return_stmt"operator">: "return" expr? stmt_end
"rule-name" id="break_stmt">break_stmt"operator">: "break" stmt_end
"rule-name" id="continue_stmt">continue_stmt"operator">: "continue" stmt_end
"rule-name" id="while_stmt">while_stmt"operator">: "while" expr block
"rule-name" id="for_stmt">for_stmt"operator">: "for" pattern "in" expr block
"rule-name" id="loop_stmt">loop_stmt"operator">: "loop" block
"rule-name" id="if_stmt">if_stmt"operator">: "if" expr block ("else" (if_stmt | block))?
"rule-name" id="expr_stmt">expr_stmt"operator">: expr stmt_end

"rule-name" id="block">block"operator">: ":" _NEWLINE _INDENT (_NEWLINE | statement)* _DEDENT | LBRACE (_NEWLINE | statement)* RBRACE

"rule-name" id="pattern">pattern"operator">: IDENT
       | IDENT ":" type_expr

"rule-name" id="type_expr">type_expr"operator">: path ("<" type_list ">")?
         | "fn" LPAREN type_list? RPAREN ("->" type_expr)?
         | "[" type_expr "]"
         | "(" type_list ")"
         | "dyn"
         | "Self"

"rule-name" id="expr">expr"operator">: logic_or (("=" | "+=" | "-=" | "*=" | "/=" | "%=") logic_or)*

"rule-name" id="?logic_or">?logic_or"operator">: logic_and ("||" logic_and)*
"rule-name" id="?logic_and">?logic_and"operator">: eq_expr ("&&" eq_expr)*
"rule-name" id="?eq_expr">?eq_expr"operator">: comp_expr (("==" | "!=") comp_expr)*
"rule-name" id="?comp_expr">?comp_expr"operator">: bit_or (("<"|"<="|">"|">=") bit_or)*
"rule-name" id="?bit_or">?bit_or"operator">: bit_xor ("|" bit_xor)*
"rule-name" id="?bit_xor">?bit_xor"operator">: bit_and ("^" bit_and)*
"rule-name" id="?bit_and">?bit_and"operator">: shift_expr ("&" shift_expr)*
"rule-name" id="?shift_expr">?shift_expr"operator">: add_expr (("<<"|">>") add_expr)*
"rule-name" id="?add_expr">?add_expr"operator">: mul_expr (("+"|"-") mul_expr)*
"rule-name" id="?mul_expr">?mul_expr"operator">: unary_expr (("*"|"/"|"%") unary_expr)*

"rule-name" id="?unary_expr">?unary_expr"operator">: ("-"|"!"|"~") unary_expr
           | "await" unary_expr
           | "try" unary_expr
           | primary

"rule-name" id="?primary">?primary"operator">: atom (postfix)*

"rule-name" id="?postfix">?postfix"operator">: "." IDENT
        | LPAREN arg_list? RPAREN
        | LBRACKET expr RBRACKET
        | "?"

"rule-name" id="arg_list">arg_list"operator">: arg ("," arg)* ","?
"rule-name" id="arg">arg"operator">: (IDENT "=")? expr

"rule-name" id="?atom">?atom"operator">: INT | FLOAT | STRING | FSTRING | RSTRING | "true" | "false" | "self"
     | path
     | "[" arg_list? "]"
     | "(" expr ")"
     | "match" expr ":" match_body
     | closure

"rule-name" id="closure">closure"operator">: "|" ident_list? "|" expr

"rule-name" id="match_body">match_body"operator">: ":" _NEWLINE _INDENT match_arm* _DEDENT | LBRACE match_arm* RBRACE
"rule-name" id="match_arm">match_arm"operator">: pattern "=>" expr ","? _NEWLINE?

"rule-name" id="path">path"operator">: IDENT ("::" IDENT)*

"terminal" id="IDENT">IDENT"operator">: /[a-zA-Z_][a-zA-Z0-9_]*/
"terminal" id="INT">INT"operator">: /[0-9]+/
"terminal" id="FLOAT">FLOAT"operator">: /[0-9]+\.[0-9]+/
"terminal" id="STRING">STRING"operator">: /"[^"]*"/ | /'[^']*'/
"terminal" id="FSTRING.2">FSTRING.2"operator">: /f"[^"]*"/ | /f'[^']*'/
"terminal" id="RSTRING.2">RSTRING.2"operator">: /r"[^"]*"/ | /r'[^']*'/

"terminal" id="LPAREN">LPAREN"operator">: "("
"terminal" id="RPAREN">RPAREN"operator">: ")"
"terminal" id="LBRACE">LBRACE"operator">: "{"
"terminal" id="RBRACE">RBRACE"operator">: "}"
"terminal" id="LBRACKET">LBRACKET"operator">: "["
"terminal" id="RBRACKET">RBRACKET"operator">: "]"

%declare _INDENT _DEDENT
"terminal" id="_NEWLINE">_NEWLINE"operator">: ( /\r?\n[\t ]*/ | /#[^\n]*\n[\t ]*/ )+
"terminal" id="COMMENT">COMMENT"operator">: "//" /[^\n]*/
"terminal" id="BLOCK_COMMENT">BLOCK_COMMENT"operator">: "/*" /(.|\n)*?/ "*/"

%ignore COMMENT
%ignore BLOCK_COMMENT
%ignore /[\t \f]+/
%ignore /#[^\n]*/
